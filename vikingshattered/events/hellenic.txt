		}

	}
	
	option = {
		name = "hellenic_events.0.a"
		trigger = { 
			NOT = { has_ruler_modifier = cynical_believer }
			NOT = { has_ruler_modifier = improving_ability }
		}
		add_ruler_modifier = { 
			name = cynical_believer
			hidden = yes
		}
		add_prestige = -2
	}
	
	option = {
