namespace = phon
country_event = {
	id = phon.1
	title = "p01name"
	desc = "p01desc"
	picture = "SHIP_SAILING"
	
	trigger = {
		NOT = { exists = P02 }
		tag = P01
		AND = {
			owns_core_province = 382
			owns_core_province = 381
		}
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	option = {
		name = "phon.1.a"
		
		381 = {
			add_core = P02
			cede_province = P02
			remove_core = P01
			change_culture = hebrew
		}
		
		382 = {
			add_core = P02
			cede_province = P02
			remove_core = P01
			change_culture = phoenician
		}
		P02 = {
			change_religion = druze
			change_primary_culture = hebrew
			add_accepted_culture = phoenician
		}
		create_subject = {
			subject = P02
			subject_type = phon_city
		}
	}
}
